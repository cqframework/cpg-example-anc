define "Gestational Age in Weeks":
  First(
    ["Observation": "Weeks of gestational age"] O
		  where O.status in { 'final', 'amended' }
      sort by FHIRHelpers.ToDateTime(effective as FHIR.dateTime) descending
  ).value as Quantity
